/**
//DIVIDER
@memberOf UI
Convert r,g,b as numbers between 0 and 256 to a string in format '#ffffff'
*/
UI.rgbStr = function(r,g,b) {
  return '#' + ((1<<24) + (r<<16) + (g<<8) + b|0).toString(16).slice(1);
};
